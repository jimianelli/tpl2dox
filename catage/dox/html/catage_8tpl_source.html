123  \ingroup PROCEDURE_SECTION
124  Function to compute mortality and survival rate. */
125 void model_parameters::get_mortality_and_survivial_rates(void)
126 {
127  int i, j;
128  // calculate the selectivity from the sel_coffs
129  for (j=1;j<nages;j++)
130  {
131  log_sel(j)=log_sel_coff(j);
132  }
133  // the selectivity is the same for the last two age classes
134  log_sel(nages)=log_sel_coff(nages-1);
135 
136  // This is the same as F(i,j)=exp(q)*effert(i)*exp(log_sel(j));
137  F=outer_prod(mfexp(log_q)*effort,mfexp(log_sel));
138  if (active(effort_devs))
139  {
140  for (i=1;i<=nyrs;i++)
141  {
142  F(i)=F(i)*exp(effort_devs(i));
143  }
144  }
145  // get the total mortality
146  Z=F+M;
147  // get the survival rate
148  S=mfexp(-1.0*Z);
149 
150 }

153  /**
154  \ingroup PROCEDURE_SECTION
155  Compute numbers at age. */
156 void model_parameters::get_numbers_at_age(void)
157 {
158  int i, j;
159  log_initpop=log_relpop+log_popscale;
160  for (i=1;i<=nyrs;i++)
161  {
162  N(i,1)=mfexp(log_initpop(i));
163  }
164  for (j=2;j<=nages;j++)
165  {
166  N(1,j)=mfexp(log_initpop(nyrs+j-1));
167  }
168  for (i=1;i<nyrs;i++)
169  {
170  for (j=1;j<nages;j++)
171  {
172  N(i+1,j+1)=N(i,j)*S(i,j);
173  }
174  }
175  // calculated predicted numbers at age for next year
176  for (j=1;j<nages;j++)
177  {
178  predicted_N(j+1)=N(nyrs,j)*S(nyrs,j);
179  ratio_N(j+1)=predicted_N(j+1)/N(1,j+1);
180  }
181  // calculated predicted Biomass for next year for
182  // adjusted profile likelihood
183  pred_B=(predicted_N * relwt);
184 
185 }

198  /**
199  \ingroup PROCEDURE_SECTION
200  Compute objective function. */
201 void model_parameters::evaluate_the_objective_function(void)
202 {
203  // penalty functions to ``regularize '' the solution
204  f+=.01*norm2(log_relpop);
205  avg_F=sum(F)/double(size_count(F));
206  if (last_phase())
207  {
208  // a very small penalty on the average fishing mortality
209  f+= .001*square(log(avg_F/.2));
210  }
211  else
212  {
213  f+= 1000.*square(log(avg_F/.2));
214  }
215  f+=0.5*double(size_count(C)+size_count(effort_devs))
216  * log( sum(elem_div(square(C-obs_catch_at_age),.01+C))
217  + 0.1*norm2(effort_devs));
218 
219 }
