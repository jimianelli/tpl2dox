137 void model_parameters::calculate_biomass_and_predicted_catch(void)
138 {
139  // calculate the biomass and predicted catch
140  if (!active(log_binit))
141  {
142  log_binit=log(obs_catch(1)/(q*effort(1)));
143  }
144  binit=exp(log_binit);
145  biomass[1]=binit;
146  if (active(k_devs))
147  {
148  k(1)=binit/beta;
149  for (int i=2;i<=nobs;i++)
150  {
151  k(i)=k(i-1)*exp(k_devs(i));
152  }
153  }
154  else
155  {
156  // set the whole vector equal to the constant k value
157  k=binit/beta;
158  }
159  // only calculate these for the standard deviation report
160  if (sd_phase())
161  {
162  k_1=k(1);
163  k_last=k(nobs);
164  k_ratio=k(nobs)/k(1);
165  k_change=k(nobs)-k(1);
166  }
167  // two time steps per year desired
168  int nsteps=2;
169  double delta=1./nsteps;
170  // Integrate the logistic dynamics over n time steps per year
171  for (int i=1; i<=nobs; i++)
172  {
173  bio_tmp=1.e-20+biomass[i];
174  c_tmp=0.;
175  for (int j=1; j<=nsteps; j++)
176  {
177  //This is the new biomass after time step delta
178  bio_tmp=bio_tmp*(1.+r*delta)/
179  (1.+ (r*pow(bio_tmp/k(i),m-1.)+f(i))*delta );
180  // This is the catch over time step delta
181  c_tmp+=f(i)*delta*bio_tmp;
182  }
183  pred_catch[i]=c_tmp; // This is the catch for this year
184  if (i<nobs)
185  {
186  biomass[i+1]=bio_tmp;// This is the biomass at the begining of the next
187  } // year
188  else
189  {
190  B_projected=bio_tmp; // get the projected biomass for std dev report
191  }
192  }
193 
194 }

196 void model_parameters::calculate_the_objective_function(void)
197 {
198  if (!active(effort_devs))
199  {
200  ff=nobs/2.*log(norm2(log(obs_catch)-log(1.e-10+pred_catch)));
201  }
202  else if(!active(k_devs))
203  {
204  ff= .5*(size_count(obs_catch)+size_count(effort_devs))*
205  log(norm2(log(obs_catch)-log(1.e-10+pred_catch))
206  +0.1*norm2(effort_devs));
207  }
208  else
209  {
210  ff= .5*( size_count(obs_catch)+size_count(effort_devs)
211  +size_count(k_devs) )*
212  log(norm2(log(obs_catch)-log(pred_catch))
213  + 0.1*norm2(effort_devs)+10.*norm2(k_devs));
214  }
215  // Bayesian contribution for Pella Tomlinson m
216  ff+=2.*square(log(m-1.));
